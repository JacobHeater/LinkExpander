How to Use the REST API 
                    
    //jQuery AJAX
    $.get("https://linkinflater.herokuapp.com/api/inflate?url=https://youtu.be/Kfpnwk-DXrA")
    .done(function(repsonse) {
        if (response.success) {
            //Everything went through without any hiccups.
            var inflatedUrl = response.data.inflatedUrl;
            //Do something with inflatedUrl
        } else {
            //Whoops! Something went wrong!
            console.error(repsonse.data);
        }
    })
    .fail(function(xhr) {
        //XmlHttpRequest error.
        console.error(xhr);
    });

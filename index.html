Link Inflater
A super simple, free, open-source REST API for expanding shortened URLs.
See the API in Action!

How to Use the REST API 
                    
    //jQuery AJAX
    $.get("https://linkinflater.herokuapp.com/api/inflate?url=https://youtu.be/Kfpnwk-DXrA")
    .done(function(repsonse) {
        if (response.success) {
            //Everything went through without any hiccups.
            var inflatedUrl = response.data.inflatedUrl;
            //Do something with inflatedUrl
        } else {
            //Whoops! Something went wrong!
            console.error(repsonse.data);
        }
    })
    .fail(function(xhr) {
        //XmlHttpRequest error.
        console.error(xhr);
    });

    //Sample Successful JSON response
    {
        "success": true,
        "data": {
            "requestedUrl": "https://youtu.be/Kfpnwk-DXrA",
            "inflatedUrl": "https://www.youtube.com/watch?v=Kfpnwk-DXrA&feature=youtu.be"
        }
    }

    //Sample Failure JSON response
    {
        "success": false,
        "data": "Error message will be here..."
    }
